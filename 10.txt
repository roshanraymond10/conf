package com.example.marksapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    private EditText etSub1, etSub2, etSub3;
    private TextView tvSum, tvAverage;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Initialize input fields
        etSub1 = findViewById(R.id.et_sub1);
        etSub2 = findViewById(R.id.et_sub2);
        etSub3 = findViewById(R.id.et_sub3);
        
        // Initialize output fields
        tvSum = findViewById(R.id.tv_sum);
        tvAverage = findViewById(R.id.tv_average);
        
        Button btnCalculate = findViewById(R.id.btn_calculate);
        
        btnCalculate.setOnClickListener(v -> calculateResults());
    }

    private void calculateResults() {
        // Read input strings
        String strSub1 = etSub1.getText().toString();
        String strSub2 = etSub2.getText().toString();
        String strSub3 = etSub3.getText().toString();

        if (strSub1.isEmpty() || strSub2.isEmpty() || strSub3.isEmpty()) {
            Toast.makeText(this, "Please enter marks for all three subjects.", Toast.LENGTH_SHORT).show();
            return;
        }

        try {
            // Convert to double (allowing for decimal marks)
            double sub1 = Double.parseDouble(strSub1);
            double sub2 = Double.parseDouble(strSub2);
            double sub3 = Double.parseDouble(strSub3);
            
            // Validate marks (optional, but good practice)
            if (sub1 < 0 || sub2 < 0 || sub3 < 0) {
                 Toast.makeText(this, "Marks cannot be negative.", Toast.LENGTH_SHORT).show();
                 return;
            }

            // Calculate Sum
            double sum = sub1 + sub2 + sub3;
            
            // Calculate Average (divided by 3 subjects)
            double average = sum / 3.0;

            // Display results, formatted to two decimal places
            tvSum.setText(String.format("Total Marks: %.2f", sum));
            tvAverage.setText(String.format("Average Marks: %.2f", average));

        } catch (NumberFormatException e) {
            Toast.makeText(this, "Invalid mark entered. Please use numbers.", Toast.LENGTH_LONG).show();
        }
    }
}

package com.example.calculatorapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    private TextView display;
    private double currentValue = 0;
    private String currentOperator = "";
    private boolean waitingForNewNumber = true;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        display = findViewById(R.id.tv_display);

        // Setup listeners for all buttons
        setupNumberButtons();
        setupOperatorButtons();
        setupFunctionButtons();
    }
    
    // Attaches listeners to digits (0-9) and decimal point
    private void setupNumberButtons() {
        int[] numberButtonIds = {R.id.btn_0, R.id.btn_1, R.id.btn_2, R.id.btn_3, R.id.btn_4, R.id.btn_5, R.id.btn_6, R.id.btn_7, R.id.btn_8, R.id.btn_9, R.id.btn_decimal};
        
        View.OnClickListener numberListener = v -> {
            Button b = (Button) v;
            String buttonText = b.getText().toString();
            
            if (waitingForNewNumber) {
                // If starting a new number or after equals
                display.setText(buttonText.equals(".") ? "0." : buttonText);
                waitingForNewNumber = false;
            } else if (buttonText.equals(".")) {
                // Prevent multiple decimals
                if (!display.getText().toString().contains(".")) {
                    display.append(".");
                }
            } else {
                // Append digits
                if (display.getText().toString().equals("0")) {
                    display.setText(buttonText);
                } else {
                    display.append(buttonText);
                }
            }
        };

        for (int id : numberButtonIds) {
            findViewById(id).setOnClickListener(numberListener);
        }
    }
    
    // Attaches listeners to +, -, *, /
    private void setupOperatorButtons() {
        int[] operatorButtonIds = {R.id.btn_add, R.id.btn_subtract, R.id.btn_multiply, R.id.btn_divide};
        
        View.OnClickListener operatorListener = v -> {
            Button b = (Button) v;
            String newOperator = b.getText().toString();
            
            if (!waitingForNewNumber && !currentOperator.isEmpty()) {
                // If chaining operations, calculate the pending one first
                performCalculation();
            } else {
                // Load the first value
                try {
                    currentValue = Double.parseDouble(display.getText().toString());
                } catch (NumberFormatException e) {
                    currentValue = 0;
                }
            }
            
            currentOperator = newOperator;
            waitingForNewNumber = true;
        };

        for (int id : operatorButtonIds) {
            findViewById(id).setOnClickListener(operatorListener);
        }
        
        // Equals button listener
        findViewById(R.id.btn_equals).setOnClickListener(v -> {
            if (!currentOperator.isEmpty()) {
                performCalculation();
                currentOperator = ""; // Reset operator after equals
                waitingForNewNumber = true;
            }
        });
    }

    // Attaches listeners to C, +/-, %
    private void setupFunctionButtons() {
        // Clear (C) button
        findViewById(R.id.btn_clear).setOnClickListener(v -> {
            display.setText("0");
            currentValue = 0;
            currentOperator = "";
            waitingForNewNumber = true;
        });
        
        // Toggle Sign (+/-) button
        findViewById(R.id.btn_toggle_sign).setOnClickListener(v -> {
            try {
                double val = Double.parseDouble(display.getText().toString());
                val = -val;
                display.setText(formatResult(val));
            } catch (NumberFormatException e) {
                // Ignore if invalid
            }
        });
        
        // Percentage (%) button
        findViewById(R.id.btn_percent).setOnClickListener(v -> {
            try {
                double val = Double.parseDouble(display.getText().toString());
                val = val / 100.0;
                display.setText(formatResult(val));
            } catch (NumberFormatException e) {
                 // Ignore if invalid
            }
        });
    }
    
    // Core calculation logic
    private void performCalculation() {
        double secondValue = 0;
        try {
            secondValue = Double.parseDouble(display.getText().toString());
        } catch (NumberFormatException e) {
            Toast.makeText(this, "Invalid number input.", Toast.LENGTH_SHORT).show();
            return;
        }

        switch (currentOperator) {
            case "+":
                currentValue += secondValue;
                break;
            case "-":
                currentValue -= secondValue;
                break;
            case "*":
                currentValue *= secondValue;
                break;
            case "/":
                if (secondValue != 0) {
                    currentValue /= secondValue;
                } else {
                    // Handle division by zero error
                    display.setText("Error");
                    currentValue = 0;
                    currentOperator = "";
                    waitingForNewNumber = true;
                    Toast.makeText(this, "Cannot divide by zero.", Toast.LENGTH_LONG).show();
                    return;
                }
                break;
            default:
                // No operator pending, just display the number
                currentValue = secondValue;
                break;
        }

        display.setText(formatResult(currentValue));
        waitingForNewNumber = true;
    }
    
    // Helper to format the result (remove unnecessary decimals for integers)
    private String formatResult(double result) {
        if (result == (long) result) {
            return String.format("%d", (long) result);
        } else {
            return String.valueOf(result);
        }
    }
}
