package com.example.datetimeapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.TextView;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.Locale;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        TextView tvFormat1 = findViewById(R.id.tv_format_1);
        TextView tvFormat2 = findViewById(R.id.tv_format_2);
        TextView tvFormat3 = findViewById(R.id.tv_format_3);

        // 1. Get current date and time
        Date now = new Date();

        // 2. Define different date/time formats
        // EEEE = Full day name, MMMM = Full month name, yyyy = 4-digit year, HH:mm = 24hr time
        SimpleDateFormat fullFormat = new SimpleDateFormat("EEEE, MMMM d, yyyy | HH:mm:ss", Locale.getDefault());
        // h:mm a = 12hr time with AM/PM
        SimpleDateFormat timeOnlyFormat = new SimpleDateFormat("h:mm:ss a", Locale.getDefault());
        // MMM = Abbreviated month, dd = day of month, yy = 2-digit year
        SimpleDateFormat dayMonthFormat = new SimpleDateFormat("MMM dd, yy", Locale.getDefault());

        // 3. Apply formats to TextViews
        tvFormat1.setText(fullFormat.format(now));
        tvFormat2.setText(timeOnlyFormat.format(now));
        tvFormat3.setText(dayMonthFormat.format(now));
    }
}

package com.example.alertdialogapp;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;
import android.content.DialogInterface;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button showDialogButton = findViewById(R.id.btn_show_dialog);
        showDialogButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                showSimpleAlertDialog();
            }
        });
    }

    private void showSimpleAlertDialog() {
        // Use AlertDialog.Builder to construct the dialog
        new AlertDialog.Builder(this)
            .setTitle("Confirm Exit")
            .setMessage("Are you sure you want to close this application?")
            .setIcon(android.R.drawable.ic_dialog_alert) // System alert icon
            
            // Set Positive button (e.g., "Yes")
            .setPositiveButton("Yes, Exit", new DialogInterface.OnClickListener() {
                public void onClick(DialogInterface dialog, int which) {
                    // Action when "Yes" is clicked
                    finish(); // Closes the activity
                    Toast.makeText(MainActivity.this, "Application Closed", Toast.LENGTH_SHORT).show();
                }
            })

            // Set Negative button (e.g., "No")
            .setNegativeButton("No, Stay", new DialogInterface.OnClickListener() {
                public void onClick(DialogInterface dialog, int which) {
                    // Action when "No" is clicked
                    dialog.dismiss(); // Closes the dialog
                    Toast.makeText(MainActivity.this, "Canceled", Toast.LENGTH_SHORT).show();
                }
            })
            
            // Set Neutral button (Optional)
            .setNeutralButton("Later", null) // 'null' closes the dialog without action
            
            .setCancelable(false) // Prevents closing on back button or outside touch
            .show(); // Display the dialog to the user
    }
}
