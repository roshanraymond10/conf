APP: Call App using Implicit Intent (ACTION_DIAL)
==============================================
NOTE: 
1. Uses ACTION_DIAL, which opens the phone app.
2. ACTION_DIAL does NOT require CALL_PHONE permission.

--- AndroidManifest.xml (Modifications) ---

    </application>

--- Functionality (MainActivity.java) ---

package com.example.callappimplicit;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import android.widget.Button;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button btnCall = findViewById(R.id.btn_call);
        btnCall.setOnClickListener(v -> launchDialApp());
    }

    private void launchDialApp() {
        String phoneNumber = "5551234567";
        
        // 1. Create the Intent with ACTION_DIAL and the 'tel' URI scheme
        Intent intent = new Intent(Intent.ACTION_DIAL);
        intent.setData(Uri.parse("tel:" + phoneNumber));
        
        // 2. Check if a suitable application exists and start the intent
        if (intent.resolveActivity(getPackageManager()) != null) {
            startActivity(intent);
        } else {
            Toast.makeText(this, "No application found to handle dialing.", Toast.LENGTH_LONG).show();
        }
    }
}

package com.example.smsappimplicit;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import android.widget.Button;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button btnSms = findViewById(R.id.btn_sms);
        btnSms.setOnClickListener(v -> launchSmsApp());
    }

    private void launchSmsApp() {
        String phoneNumber = "5557654321";
        String message = "Hello from my Android app!";
        
        // 1. Create the Intent with ACTION_SENDTO and the 'smsto' URI scheme
        Intent intent = new Intent(Intent.ACTION_SENDTO);
        intent.setData(Uri.parse("smsto:" + phoneNumber)); 
        
        // 2. Add the message body as an Extra
        intent.putExtra("sms_body", message);
        
        // 3. Check if a suitable application exists and start the intent
        if (intent.resolveActivity(getPackageManager()) != null) {
            startActivity(intent);
        } else {
            Toast.makeText(this, "No SMS app found to handle this request.", Toast.LENGTH_LONG).show();
        }
    }
}

package com.example.emailappimplicit;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import android.widget.Button;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button btnEmail = findViewById(R.id.btn_email);
        btnEmail.setOnClickListener(v -> launchEmailApp());
    }

    private void launchEmailApp() {
        Intent intent = new Intent(Intent.ACTION_SENDTO);
        
        // 1. Set data to 'mailto:' scheme to ensure only email apps respond
        intent.setData(Uri.parse("mailto:")); 
        
        // 2. Add email details as Extras
        intent.putExtra(Intent.EXTRA_EMAIL, new String[]{"recipient@example.com", "cc@example.com"});
        intent.putExtra(Intent.EXTRA_SUBJECT, "Feedback from App");
        intent.putExtra(Intent.EXTRA_TEXT, "Body of the email goes here.");
        
        // 3. Check if a suitable application exists and start the intent
        if (intent.resolveActivity(getPackageManager()) != null) {
            startActivity(intent);
        } else {
            Toast.makeText(this, "No email client found to handle this request.", Toast.LENGTH_LONG).show();
        }
    }
}

package com.example.listsensorsapp;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Context;
import android.hardware.Sensor;
import android.hardware.SensorManager;
import android.os.Bundle;
import android.widget.ArrayAdapter;
import android.widget.ListView;
import java.util.ArrayList;
import java.util.List;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        ListView sensorListView = findViewById(R.id.sensor_list_view);
        
        // 1. Get the SensorManager service
        SensorManager sensorManager = (SensorManager) getSystemService(Context.SENSOR_SERVICE);
        
        // 2. Get the list of ALL sensors
        List<Sensor> deviceSensors = sensorManager.getSensorList(Sensor.TYPE_ALL);

        // 3. Prepare data for the ListView
        ArrayList<String> sensorInfo = new ArrayList<>();
        for (Sensor sensor : deviceSensors) {
            String info = "Name: " + sensor.getName() + 
                          "\nVendor: " + sensor.getVendor() + 
                          "\nType: " + sensor.getStringType();
            sensorInfo.add(info);
        }

        // 4. Populate the ListView
        ArrayAdapter<String> adapter = new ArrayAdapter<>(
            this,
            android.R.layout.simple_list_item_1, 
            sensorInfo
        );
        sensorListView.setAdapter(adapter);
    }
}
