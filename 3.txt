package com.example.radiocheckboxapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.RadioGroup;
import android.widget.RadioButton;
import android.widget.CheckBox;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    private RadioGroup radioGroup;
    private CheckBox checkFlutter, checkReact, checkXamarin;
    private TextView resultsText;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Find Views
        radioGroup = findViewById(R.id.radio_group_language);
        checkFlutter = findViewById(R.id.check_flutter);
        checkReact = findViewById(R.id.check_react);
        checkXamarin = findViewById(R.id.check_xamarin);
        Button submitButton = findViewById(R.id.btn_submit);
        resultsText = findViewById(R.id.tv_results);

        submitButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                displayChoices();
            }
        });
    }

    private void displayChoices() {
        // --- 1. Get Radio Button Choice (Single Selection) ---
        int selectedId = radioGroup.getCheckedRadioButtonId();
        String primaryLang = "No language selected";

        if (selectedId != -1) {
            RadioButton selectedRadio = findViewById(selectedId);
            primaryLang = selectedRadio.getText().toString();
        }

        // --- 2. Get Checkbox Choices (Multiple Selection) ---
        StringBuilder frameworks = new StringBuilder("Frameworks: ");

        if (checkFlutter.isChecked()) {
            frameworks.append("Flutter, ");
        }
        if (checkReact.isChecked()) {
            frameworks.append("React Native, ");
        }
        if (checkXamarin.isChecked()) {
            frameworks.append("Xamarin, ");
        }
        
        // Clean up the comma/space if any frameworks were selected
        String frameworkString = frameworks.toString();
        if (frameworkString.endsWith(", ")) {
            frameworkString = frameworkString.substring(0, frameworkString.length() - 2);
        } else if (frameworkString.equals("Frameworks: ")) {
            frameworkString = "No frameworks selected";
        }


        // --- 3. Display Results ---
        String finalResult = "Primary Language: " + primaryLang + "\n" + frameworkString;
        resultsText.setText(finalResult);
        Toast.makeText(this, "Choices Submitted!", Toast.LENGTH_SHORT).show();
    }
}

package com.example.sumapp;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    public static final String EXTRA_SUM = "com.example.sumapp.SUM"; // Key for the Intent Extra

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        final EditText etNum1 = findViewById(R.id.et_num1);
        final EditText etNum2 = findViewById(R.id.et_num2);
        Button btnCalculate = findViewById(R.id.btn_calculate);

        btnCalculate.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Get user input as strings
                String strNum1 = etNum1.getText().toString();
                String strNum2 = etNum2.getText().toString();
                
                if (strNum1.isEmpty() || strNum2.isEmpty()) {
                    Toast.makeText(MainActivity.this, "Please enter both numbers", Toast.LENGTH_SHORT).show();
                    return;
                }

                try {
                    // Convert to integers and calculate sum
                    int num1 = Integer.parseInt(strNum1);
                    int num2 = Integer.parseInt(strNum2);
                    int sum = num1 + num2;

                    // Create Intent to launch ResultActivity
                    Intent intent = new Intent(MainActivity.this, ResultActivity.class);
                    
                    // Add the calculated sum as an Extra
                    intent.putExtra(EXTRA_SUM, sum); 
                    
                    // Start the new activity
                    startActivity(intent);
                } catch (NumberFormatException e) {
                    Toast.makeText(MainActivity.this, "Invalid number format", Toast.LENGTH_SHORT).show();
                }
            }
        });
    }
}

--- Functionality (ResultActivity.java) ---

package com.example.sumapp;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.os.Bundle;
import android.widget.TextView;

public class ResultActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_result);

        TextView tvResult = findViewById(R.id.tv_result);

        // Get the Intent that started this activity
        Intent intent = getIntent();
        
        // Retrieve the integer 'Extra' using the key defined in MainActivity
        // The '0' is the default value if the Extra is not found.
        int sumResult = intent.getIntExtra(MainActivity.EXTRA_SUM, 0);

        // Display the result
        tvResult.setText(String.valueOf(sumResult));
    }
}

package com.example.launchwebpageapp;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    private static final String URL_TO_OPEN = "https://www.google.com";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button launchButton = findViewById(R.id.btn_launch_web);

        launchButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // 1. Create a URI object from the URL string
                Uri webpage = Uri.parse(URL_TO_OPEN);
                
                // 2. Create an Implicit Intent
                Intent intent = new Intent(Intent.ACTION_VIEW, webpage);

                // 3. Check if there is an app that can handle this Intent (e.g., a browser)
                if (intent.resolveActivity(getPackageManager()) != null) {
                    startActivity(intent);
                } else {
                    Toast.makeText(MainActivity.this, "No browser app found to handle this request.", Toast.LENGTH_LONG).show();
                }
            }
        });
    }
}
