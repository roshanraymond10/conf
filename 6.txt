package com.example.simplecounterapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.Button;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    private TextView countTextView;
    private int currentCount = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        countTextView = findViewById(R.id.tv_count);
        Button incrementButton = findViewById(R.id.btn_increment);
        Button resetButton = findViewById(R.id.btn_reset);

        // Set initial count
        updateCountDisplay();

        // Increment Button Logic
        incrementButton.setOnClickListener(v -> {
            currentCount++;
            updateCountDisplay();
        });

        // Reset Button Logic
        resetButton.setOnClickListener(v -> {
            currentCount = 0;
            updateCountDisplay();
        });
    }

    private void updateCountDisplay() {
        countTextView.setText(String.valueOf(currentCount));
    }
}

package com.example.texttospeechapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.speech.tts.TextToSpeech;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;
import java.util.Locale;

public class MainActivity extends AppCompatActivity implements TextToSpeech.OnInitListener {

    private TextToSpeech tts;
    private EditText inputText;
    private Button speakButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        inputText = findViewById(R.id.et_input_text);
        speakButton = findViewById(R.id.btn_speak);
        
        // 1. Initialize the TextToSpeech engine
        tts = new TextToSpeech(this, this);

        // 2. Set click listener for the speak button
        speakButton.setOnClickListener(v -> speakOut());
    }

    // Called when the TTS engine is initialized
    @Override
    public void onInit(int status) {
        if (status == TextToSpeech.SUCCESS) {
            // 3. Set the language
            int result = tts.setLanguage(Locale.US);

            if (result == TextToSpeech.LANG_MISSING_DATA || result == TextToSpeech.LANG_NOT_SUPPORTED) {
                Toast.makeText(this, "Language is not supported on this device.", Toast.LENGTH_LONG).show();
                speakButton.setEnabled(false);
            } else {
                speakButton.setEnabled(true);
            }
        } else {
            Toast.makeText(this, "TTS Initialization Failed!", Toast.LENGTH_LONG).show();
            speakButton.setEnabled(false);
        }
    }

    private void speakOut() {
        String text = inputText.getText().toString().trim();
        if (text.isEmpty()) {
            text = "Please enter some text to speak.";
        }
        
        // Check if TTS engine is ready before speaking
        if (tts != null && speakButton.isEnabled()) {
            // Set queue mode to QUEUE_FLUSH to stop previous speech and start new one immediately
            tts.speak(text, TextToSpeech.QUEUE_FLUSH, null, "unique_id");
        }
    }

    // 4. Release resources when the activity is destroyed
    @Override
    public void onDestroy() {
        if (tts != null) {
            tts.stop();
            tts.shutdown();
        }
        super.onDestroy();
    }
}

package com.example.studentmanagement;

public class Student {
    private int id;
    private String name;
    private int age;

    public Student(int id, String name, int age) {
        this.id = id;
        this.name = name;
        this.age = age;
    }
    
    // Getters
    public int getId() { return id; }
    public String getName() { return name; }
    public int getAge() { return age; }

    // Setters (useful for updates)
    public void setName(String name) { this.name = name; }
    public void setAge(int age) { this.age = age; }

    @Override
    public String toString() {
        return "ID: " + id + ", Name: " + name + ", Age: " + age;
    }
}

--- Database Helper (StudentDBHelper.java) ---

package com.example.studentmanagement;

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;
import java.util.ArrayList;
import java.util.List;

public class StudentDBHelper extends SQLiteOpenHelper {

    private static final String DATABASE_NAME = "students.db";
    private static final int DATABASE_VERSION = 1;
    public static final String TABLE_STUDENTS = "students_table";
    public static final String COLUMN_ID = "ID";
    public static final String COLUMN_NAME = "NAME";
    public static final String COLUMN_AGE = "AGE";

    public StudentDBHelper(Context context) {
        super(context, DATABASE_NAME, null, DATABASE_VERSION);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        String CREATE_TABLE = "CREATE TABLE " + TABLE_STUDENTS + " (" +
                COLUMN_ID + " INTEGER PRIMARY KEY AUTOINCREMENT, " +
                COLUMN_NAME + " TEXT, " +
                COLUMN_AGE + " INTEGER)";
        db.execSQL(CREATE_TABLE);
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_STUDENTS);
        onCreate(db);
    }

    // --- C (Create / Add) ---
    public boolean addStudent(Student student) {
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues cv = new ContentValues();
        cv.put(COLUMN_NAME, student.getName());
        cv.put(COLUMN_AGE, student.getAge());
        long insert = db.insert(TABLE_STUDENTS, null, cv);
        db.close();
        return insert != -1;
    }

    // --- R (Read / Get All) ---
    public List<Student> getAllStudents() {
        List<Student> studentList = new ArrayList<>();
        SQLiteDatabase db = this.getReadableDatabase();
        Cursor cursor = db.rawQuery("SELECT * FROM " + TABLE_STUDENTS, null);

        if (cursor.moveToFirst()) {
            do {
                int id = cursor.getInt(0);
                String name = cursor.getString(1);
                int age = cursor.getInt(2);
                studentList.add(new Student(id, name, age));
            } while (cursor.moveToNext());
        }
        cursor.close();
        db.close();
        return studentList;
    }

    // --- U (Update) ---
    public boolean updateStudent(Student student) {
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues cv = new ContentValues();
        cv.put(COLUMN_NAME, student.getName());
        cv.put(COLUMN_AGE, student.getAge());

        // Update the row where ID matches the student's ID
        int rowsAffected = db.update(TABLE_STUDENTS, cv, COLUMN_ID + " = ?",
                new String[]{String.valueOf(student.getId())});
        db.close();
        return rowsAffected > 0;
    }

    // --- D (Delete) ---
    public boolean deleteStudent(int studentId) {
        SQLiteDatabase db = this.getWritableDatabase();
        // Delete the row where ID matches the studentId
        int rowsAffected = db.delete(TABLE_STUDENTS, COLUMN_ID + " = ?",
                new String[]{String.valueOf(studentId)});
        db.close();
        return rowsAffected > 0;
    }
}

package com.example.studentmanagement;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ListView;
import android.widget.Toast;

import java.util.List;

public class MainActivity extends AppCompatActivity {

    private EditText etId, etName, etAge;
    private Button btnAdd, btnUpdate, btnDelete;
    private ListView studentListView;
    private StudentDBHelper dbHelper;
    private ArrayAdapter<Student> studentAdapter;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Initialize views
        etId = findViewById(R.id.et_id);
        etName = findViewById(R.id.et_name);
        etAge = findViewById(R.id.et_age);
        btnAdd = findViewById(R.id.btn_add);
        btnUpdate = findViewById(R.id.btn_update);
        btnDelete = findViewById(R.id.btn_delete);
        studentListView = findViewById(R.id.student_list_view);

        dbHelper = new StudentDBHelper(this);

        // Load initial data
        showStudentsOnListView();

        // Set click listeners
        btnAdd.setOnClickListener(v -> addStudent());
        btnUpdate.setOnClickListener(v -> updateStudent());
        btnDelete.setOnClickListener(v -> deleteStudent());
        
        // Optional: Populate fields when an item is clicked for easy Update/Delete
        studentListView.setOnItemClickListener((parent, view, position, id) -> {
            Student selectedStudent = (Student) parent.getItemAtPosition(position);
            etId.setText(String.valueOf(selectedStudent.getId()));
            etName.setText(selectedStudent.getName());
            etAge.setText(String.valueOf(selectedStudent.getAge()));
        });
    }

    private void showStudentsOnListView() {
        List<Student> allStudents = dbHelper.getAllStudents();
        studentAdapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, allStudents);
        studentListView.setAdapter(studentAdapter);
        Toast.makeText(this, "Total Students: " + allStudents.size(), Toast.LENGTH_SHORT).show();
    }

    private void addStudent() {
        String name = etName.getText().toString();
        int age;
        try { age = Integer.parseInt(etAge.getText().toString()); } 
        catch (Exception e) { Toast.makeText(this, "Invalid Age", Toast.LENGTH_SHORT).show(); return; }

        Student newStudent = new Student(0, name, age); // ID is 0, will be auto-generated
        if (dbHelper.addStudent(newStudent)) {
            Toast.makeText(this, name + " added successfully!", Toast.LENGTH_SHORT).show();
            clearFields();
            showStudentsOnListView();
        } else {
            Toast.makeText(this, "Failed to add student.", Toast.LENGTH_SHORT).show();
        }
    }
    
    private void updateStudent() {
        int id;
        try { id = Integer.parseInt(etId.getText().toString()); } 
        catch (Exception e) { Toast.makeText(this, "Please enter a valid ID to update.", Toast.LENGTH_SHORT).show(); return; }
        
        String name = etName.getText().toString();
        int age;
        try { age = Integer.parseInt(etAge.getText().toString()); } 
        catch (Exception e) { Toast.makeText(this, "Invalid Age", Toast.LENGTH_SHORT).show(); return; }
        
        Student studentToUpdate = new Student(id, name, age);
        if (dbHelper.updateStudent(studentToUpdate)) {
            Toast.makeText(this, "Student ID " + id + " updated.", Toast.LENGTH_SHORT).show();
            clearFields();
            showStudentsOnListView();
        } else {
            Toast.makeText(this, "Update failed. ID not found or invalid data.", Toast.LENGTH_SHORT).show();
        }
    }

    private void deleteStudent() {
        int id;
        try { id = Integer.parseInt(etId.getText().toString()); } 
        catch (Exception e) { Toast.makeText(this, "Please enter a valid ID to delete.", Toast.LENGTH_SHORT).show(); return; }

        if (dbHelper.deleteStudent(id)) {
            Toast.makeText(this, "Student ID " + id + " deleted.", Toast.LENGTH_SHORT).show();
            clearFields();
            showStudentsOnListView();
        } else {
            Toast.makeText(this, "Delete failed. ID not found.", Toast.LENGTH_SHORT).show();
        }
    }

    private void clearFields() {
        etId.setText("");
        etName.setText("");
        etAge.setText("");
    }
}

package com.example.asynctaskdemo;

import androidx.appcompat.app.AppCompatActivity;
import android.os.AsyncTask;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.ProgressBar;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    private TextView statusTextView;
    private ProgressBar progressBar;
    private Button startButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        statusTextView = findViewById(R.id.tv_status);
        progressBar = findViewById(R.id.progress_bar);
        startButton = findViewById(R.id.btn_start_task);

        startButton.setOnClickListener(v -> {
            // Start the AsyncTask
            new DownloadTask().execute(10); // Execute with a parameter (simulated duration)
        });
    }

    // <Params, Progress, Result>
    private class DownloadTask extends AsyncTask<Integer, Integer, String> {

        // Runs on the UI thread before doInBackground
        @Override
        protected void onPreExecute() {
            super.onPreExecute();
            statusTextView.setText("Task Started...");
            progressBar.setProgress(0);
            startButton.setEnabled(false); // Disable button during task
        }

        // Runs on a background thread. Performs the long operation.
        @Override
        protected String doInBackground(Integer... params) {
            int durationSeconds = params[0];
            
            // Simulate a background task (e.g., file download)
            for (int i = 0; i <= 100; i += 10) {
                try {
                    // Sleep for a short time to simulate work
                    Thread.sleep(durationSeconds * 100); 
                } catch (InterruptedException e) {
                    return "Task Interrupted!";
                }
                
                // Publish progress to trigger onProgressUpdate
                publishProgress(i);
            }

            return "Download Complete!";
        }

        // Runs on the UI thread. Updates UI with progress from the background.
        @Override
        protected void onProgressUpdate(Integer... progress) {
            super.onProgressUpdate(progress);
            int currentProgress = progress[0];
            progressBar.setProgress(currentProgress);
            statusTextView.setText("Downloading... " + currentProgress + "%");
        }

        // Runs on the UI thread after doInBackground finishes. Receives the result.
        @Override
        protected void onPostExecute(String result) {
            super.onPostExecute(result);
            statusTextView.setText(result);
            progressBar.setProgress(100);
            startButton.setEnabled(true); // Re-enable button
            Toast.makeText(MainActivity.this, result, Toast.LENGTH_SHORT).show();
        }
    }
}
